----------------------------------------------------------
create table DBComponent_With_Status(
		DBCSId varchar(50) constraint CWSPK primary key,
		DBComponentName varchar(50),
		DBStatus varchar(50) constraint CWSCHKStatus check(DBStatus in('Operational','Minor','Major')),

	)
